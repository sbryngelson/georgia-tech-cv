% Bold names of students and underline the candidate's name
\newboolean{bold}
\newboolean{iscand}
\newcommand{\candidatename}{}
\newcommand{\makeauthorunderline}[1]{\renewcommand{\candidatename}{#1}}
\newcommand{\formatname}[1]{%
  \ifthenelse{\boolean{iscand}}{\underline{#1}}{%
    \expandafter\ifthenelse{\boolean{bold}}{\mkbibbold{#1}}{#1}}}
\newcommand{\makeauthorsbold}[1]{%
  \DeclareNameFormat{author}{%
  \setboolean{bold}{false}%
  \setboolean{iscand}{false}%
    \renewcommand{\do}[1]{\expandafter\ifstrequal\expandafter{\namepartfamily}{####1}{\setboolean{bold}{true}}{}}%
    \docsvlist{#1}%
    \begingroup\edef\tempcand{\endgroup\noexpand\ifstrequal{\namepartfamily}{\candidatename}}%
    \tempcand{\setboolean{iscand}{true}}{}%
    \ifthenelse{\value{listcount}=1}
    {%
     {\formatname{\namepartfamily\addcomma\addspace \namepartgiveni}}%
    }{\ifnumless{\value{listcount}}{\value{liststop}}
      {\addcomma\addspace\formatname{\namepartfamily\addcomma\addspace \namepartgiveni}}%
      {\addcomma\addspace\formatname{\namepartfamily\addcomma\addspace \namepartgiveni\addcomma\isdot}}%
      }
    \ifthenelse{\value{listcount}<\value{liststop}}
    {\addcomma\space}{}
  }
}

% Get reverse numbering for entries
\makeatletter
\patchcmd{\blx@printbibliography}
  {\blx@bibliography\blx@tempa}
  {\setcounter{bibitemtotal}{0}%
   \begingroup
   \def\do##1{\stepcounter{bibitemtotal}}%
   \dolistloop{\blx@tempa}%
   \endgroup
   \blx@bibliography\blx@tempa}{}{}
\makeatother
